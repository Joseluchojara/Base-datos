
-- SISTEMA QUE PERMITA INGRESANDO DE MANERA PARAMETRICA EL ID DEL EMPLEADO, NOS MUESTRE POR
--SALIDA DE CONSOLA, EL NOMBRE(NOMBRE Y APELLIDO),SUELDO Y EL NUMERO DE TELEFONO DEL
--EMPLEADO

DECLARE
    V_NOMBRE VARCHAR(15);
    V_SUELDO NUMBER(9);
    V_TELEFONO EMPLOYEES.PHONE_NUMBER%TYPE;
BEGIN
        SELECT FIRST_NAME ||' '|| LAST_NAME,SALARY,PHONE_NUMBER
        INTO V_NOMBRE, V_SUELDO, V_TELEFONO
        FROM EMPLOYEES
        WHERE EMPLOYEE_ID = &ID;
        DBMS_OUTPUT.PUT_LINE('NOMBRE:'||V_NOMBRE);
        DBMS_OUTPUT.PUT_LINE('SUELDO: '||V_SUELDO);
        DBMS_OUTPUT.PUT_LINE('TELEFONO: '||V_TELEFONO);
END;
