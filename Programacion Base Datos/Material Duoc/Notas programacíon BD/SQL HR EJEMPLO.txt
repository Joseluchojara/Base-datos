    v_VARIABLE_EXTERNA VARCHAR(20):= 'ESTA ES UNA VARIABLE INTERNA';

    DECLARE
        V_VARIABLE_INTERNA VARCHAR(20) := 'ESTA ES UNA VARIBLE INTERNA';

    END; --FIN DEL BLOQUE ANONIMO
    
    
    
    
END; -- FIN DEL BLOQUE ANONIMO PRINCIPAL
