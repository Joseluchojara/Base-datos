create or replace NONEDITIONABLE FUNCTION YEARS_BEWTWEEN (V_FECHA1 DATE, V_FECHA2 DATE)
RETURN NUMBER
IS 
    V_ANIO1 NUMBER;
    V_ANIO2 NUMBER;
    V_DIF_ANIOS NUMBER;

BEGIN


    V_ANIO1 := EXTRACT(YEAR FROM V_FECHA1);
    V_ANIO2 := EXTRACT(YEAR FROM V_FECHA2);

    V_DIF_ANIOS := V_ANIO1-V_ANIO2;

    RETURN V_DIF_ANIOS;

END;
