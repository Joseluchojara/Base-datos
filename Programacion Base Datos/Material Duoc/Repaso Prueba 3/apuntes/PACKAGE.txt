create or replace NONEDITIONABLE PACKAGE PKG_RRHH
IS  
    FUNCTION NOMBRE_MANAGER(V_ID_EMPLEADO NUMBER) RETURN VARCHAR;
    FUNCTION SUELDO_ANUAL (V_ID_EMPLEADO NUMBER) RETURN NUMBER;
    PROCEDURE AUMENTO_SUELDO_ANUAL;

END PKG_RRHH;
------------------------

create or replace NONEDITIONABLE PACKAGE BODY PKG_RRHH
IS
    FUNCTION NOMBRE_MANAGER(V_ID_EMPLEADO NUMBER) RETURN VARCHAR
    IS
        V_NOMBRE_JEFE VARCHAR(100);

    BEGIN
        SELECT JEFES.FIRST_NAME 
        INTO V_NOMBRE_JEFE
        FROM EMPLOYEES EMP
        JOIN EMPLOYEES JEFES ON EMP.MANAGER_ID = JEFES.EMPLOYEE_ID
        WHERE EMP.EMPLOYEE_ID = V_ID_EMPLEADO;

    END NOMBRE_MANAGER;
    -------------------
    FUNCTION SUELDO_ANUAL (V_ID_EMPLEADO NUMBER) RETURN NUMBER
    IS
         V_SUELDO_ANUAL NUMBER;

    BEGIN
        SELECT SALARY*12
        INTO V_SUELDO_ANUAL
        FROM EMPLOYEES
        WHERE EMPLOYEE_ID = V_ID_EMPLEADO;

        V_SUELDO_ANUAL := V_SUELDO_ANUAL+1500;

        RETURN V_SUELDO_ANUAL;

    END SUELDO_ANUAL;
    ----------------------

   PROCEDURE AUMENTO_SUELDO_ANUAL
    IS


    BEGIN
        UPDATE EMPLOYEES
        SET SALARY = SALARY + 1500;

    END AUMENTO_SUELDO_ANUAL;
END PKG_RRHH;
